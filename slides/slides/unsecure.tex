\begin{frame}[fragile]{The Unsecure module}
%\vspace{-1.5cm}
\begin{itemize}
	\item<1-> Ensures an input value will be validated before its use
	\item<1-> Is defined as a pair (Validation Functions, input value)
\end{itemize}	
\begin{center}
\vspace{1.5cm}
\begin{lstlisting}[basicstyle=\fontsize{9}{9}\ttfamily]
getNatural :: IO (Unsecure String NumericError)
getNatural = do n <- getLine
                return $ upure n [isNumeric,
                                  isPositive,
                                  isNonDecimal]
\end{lstlisting}
\end{center}
