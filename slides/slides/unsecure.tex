\begin{lstlisting}[basicstyle=\fontsize{9}{9}\ttfamily]
getSafeString :: Int -> IO (Unsecure String Bool)
getSafeString n = do l <- getLine
                     return $ upure [(\s -> length s <= n)]
\end{lstlisting}
